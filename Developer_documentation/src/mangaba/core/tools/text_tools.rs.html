7fn recursive_split(text: &str, chunk_size: usize, overlap: usize, separators: &[&str]) -> Vec<String> {
8    if text.len() <= chunk_size {
9        let trimmed = text.trim();
10        return if trimmed.is_empty() { vec![] } else { vec![trimmed.to_string()] };
11    }
12
13    let sep = separators.first().copied().unwrap_or("");
14    let remaining_seps = if separators.len() > 1 { &separators[1..] } else { &[""] };
15
16    let parts: Vec<&str> = if sep.is_empty() {
17        vec![text]
18    } else {
19        text.split(sep).collect()
20    };
21
22    let mut chunks: Vec<String> = Vec::new();
23    let mut current = String::new();
24
25    for part in parts {
26        let candidate = if current.is_empty() {
27            part.to_string()
28        } else {
29            format!("{}{}{}", current, sep, part)
30        };
31
32        if candidate.len() <= chunk_size {
33            current = candidate;
34        } else {
35            if !current.is_empty() {
36                if current.len() <= chunk_size {
37                    chunks.push(current.trim().to_string());
38                } else {
39                    chunks.extend(recursive_split(&current, chunk_size, overlap, remaining_seps));
40                }
41            }
42            current = part.to_string();
43        }
44    }
45
46    if !current.trim().is_empty() {
47        if current.len() <= chunk_size {
48            chunks.push(current.trim().to_string());
49        } else {
50            chunks.extend(recursive_split(&current, chunk_size, overlap, remaining_seps));
51        }
52    }
53
54    // Apply overlap
55    if overlap > 0 && chunks.len() > 1 {
56        let mut overlapped: Vec<String> = vec![chunks[0].clone()];
57        for i in 1..chunks.len() {
58            let prev = &chunks[i - 1];
59            let overlap_text = if prev.len() > overlap {
60                prev[prev.len().saturating_sub(overlap)..].to_string()
61            } else {
62                prev.clone()
63            };
64            overlapped.push(format!("{}{}", overlap_text, chunks[i]));
65        }
66        return overlapped;
67    }
68
69    chunks
70}

72pub struct TextSplitterTool;
73
74#[async_trait]
75impl BaseTool for TextSplitterTool {
76    fn name(&self) -> &str { "text_splitter" }
77    fn description(&self) -> &str { "Split long text into smaller chunks with optional overlap" }
78    fn args_schema(&self) -> Option<Value> {
79        Some(json!({
80            "type": "object",
81            "properties": {
82                "text": {"type": "string", "description": "The text to split into chunks"},
83                "chunk_size": {"type": "integer", "description": "Maximum characters per chunk", "default": 1000},
84                "chunk_overlap": {"type": "integer", "description": "Overlap between consecutive chunks", "default": 200}
85            },
86            "required": ["text"]
87        }))
88    }
89
90    async fn run_impl(&self, args: Value) -> Result<ToolResult> {
91        let text = args.get("text")
92            .and_then(|v| v.as_str())
93            .ok_or_else(|| anyhow!("Missing 'text' argument"))?;
94        let chunk_size = args.get("chunk_size").and_then(|v| v.as_u64()).unwrap_or(1000) as usize;
95        let chunk_overlap = args.get("chunk_overlap").and_then(|v| v.as_u64()).unwrap_or(200) as usize;
96
97        let separators = &["\n\n", "\n", ". ", " ", ""];
98        let chunks = recursive_split(text, chunk_size, chunk_overlap, separators);
99
100        let result: String = chunks.iter().enumerate()
101            .map(|(i, c)| format!("--- Chunk {}/{} ---\n{}", i + 1, chunks.len(), c))
102            .collect::<Vec<_>>()
103            .join("\n\n");
104
105        Ok(ToolResult {
106            call_id: "split".to_string(),
107            tool_name: "text_splitter".to_string(),
108            output: Some(json!({"chunks": chunks, "result": result})),
109            error: None,
110            success: true,
111        })
112    }
113}

115pub struct WordCounterTool;
116
117#[async_trait]
118impl BaseTool for WordCounterTool {
119    fn name(&self) -> &str { "word_counter" }
120    fn description(&self) -> &str { "Count words, sentences, and characters in text" }
121    fn args_schema(&self) -> Option<Value> {
122        Some(json!({
123            "type": "object",
124            "properties": {
125                "text": {"type": "string", "description": "Text to count words in"}
126            },
127            "required": ["text"]
128        }))
129    }
130
131    async fn run_impl(&self, args: Value) -> Result<ToolResult> {
132        let text = args.get("text")
133            .and_then(|v| v.as_str())
134            .ok_or_else(|| anyhow!("Missing 'text' argument"))?;
135
136        let words = text.split_whitespace().count();
137        let chars = text.chars().count();
138        let sentences = text.split(|c| c == '.' || c == '!' || c == '?')
139            .filter(|s| !s.trim().is_empty())
140            .count();
141        let paragraphs = text.split("\n\n")
142            .filter(|p| !p.trim().is_empty())
143            .count();
144
145        let result = format!(
146            "Words: {}\nCharacters: {}\nSentences: {}\nParagraphs: {}",
147            words, chars, sentences, paragraphs
148        );
149
150        Ok(ToolResult {
151            call_id: "wc".to_string(),
152            tool_name: "word_counter".to_string(),
153            output: Some(json!({
154                "words": words,
155                "characters": chars,
156                "sentences": sentences,
157                "paragraphs": paragraphs,
158                "result": result
159            })),
160            error: None,
161            success: true,
162        })
163    }
