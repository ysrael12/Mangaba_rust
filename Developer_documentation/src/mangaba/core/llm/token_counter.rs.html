1//! Token counting and usage tracking for LLM calls.
2//!
3//! [`TokenCounter`] provides static estimation methods. [`UsageTracker`] wraps
4//! an `Arc<Mutex<>>` around accumulated [`TokenUsage`]
5//! and exposes `record()` / `total()` / `reset()`.
6
7use std::sync::{Arc, Mutex};
8use crate::core::types::TokenUsage;
9
10pub struct TokenCounter;
11
12impl TokenCounter {
13    /// Heuristic: ~4 characters per token for general text
14    pub fn count_text(text: &str) -> usize {
15        let len = text.len();
16        if len == 0 { return 0; }
17        (len + 3) / 4
18    }
19
20    pub fn count_messages(messages: &[crate::core::types::Message]) -> usize {
21        messages.iter().map(|m| {
22            let content_tokens = m.content.as_deref().map(Self::count_text).unwrap_or(0);
23            let tool_tokens = m.tool_calls.as_ref()
24                .map(|calls| calls.len() * 10)
25                .unwrap_or(0);
26            let result_tokens = m.tool_results.as_ref()
27                .map(|r| r.iter().map(|tr| {
28                    tr.output.as_ref().map(|o| Self::count_text(&o.to_string())).unwrap_or(0)
29                        + tr.error.as_ref().map(|e| Self::count_text(e)).unwrap_or(0)
30                }).sum::<usize>())
31                .unwrap_or(0);
32            content_tokens + tool_tokens + result_tokens + 4 // overhead per message
33        }).sum()
34    }
35}
36
37pub struct UsageTracker {
38    usage: Arc<Mutex<TokenUsage>>,
39}
40
41impl UsageTracker {
42    pub fn new() -> Self {
43        Self { usage: Arc::new(Mutex::new(TokenUsage::default())) }
44    }
45
46    pub fn record(&self, usage: &TokenUsage) {
47        let mut u = self.usage.lock().unwrap();
48        u.prompt_tokens += usage.prompt_tokens;
49        u.completion_tokens += usage.completion_tokens;
50        u.total_tokens += usage.total_tokens;
51    }
52
53    pub fn total(&self) -> TokenUsage {
54        self.usage.lock().unwrap().clone()
55    }
56
57    pub fn reset(&self) {
58        let mut u = self.usage.lock().unwrap();
59        *u = TokenUsage::default();
60    }
61}
