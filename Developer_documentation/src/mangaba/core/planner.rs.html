1//! LLM-generated task planning and execution plans.
2//!
3//! [`TaskPlanner`] asks an LLM to decompose a task into [`PlanStep`]s
4//! (each with a description, optional tool, expected result, and dependencies).
5//! The result is an [`ExecutionPlan`] that can be executed step-by-step.
6
7use anyhow::{Result, anyhow};
8use serde::{Deserialize, Serialize};
9
10use crate::core::llm::LLMClient;
11use crate::core::types::Message;
12
13// ---------------------------------------------------------------------------
14// PlanStep
15// ---------------------------------------------------------------------------
16#[derive(Debug, Clone, Serialize, Deserialize)]
17pub struct PlanStep {
18    pub step_number: usize,
19    pub description: String,
20    #[serde(default)]
21    pub tool: Option<String>,
22    #[serde(default)]
23    pub expected_result: String,
24    #[serde(default)]
25    pub dependencies: Vec<usize>,
26}
27
28// ---------------------------------------------------------------------------
29// ExecutionPlan
30// ---------------------------------------------------------------------------
31#[derive(Debug, Clone, Serialize, Deserialize)]
32pub struct ExecutionPlan {
33    pub goal: String,
34    pub steps: Vec<PlanStep>,
35}
36
37impl ExecutionPlan {
38    pub fn total_steps(&self) -> usize {
39        self.steps.len()
40    }
41}
42
43// ---------------------------------------------------------------------------
44// TaskPlanner
45// ---------------------------------------------------------------------------
46pub struct TaskPlanner {
47    llm: Box<dyn LLMClient + Send + Sync>,
48    tool_names: Vec<String>,
49}
50
51impl TaskPlanner {
52    pub fn new(llm: Box<dyn LLMClient + Send + Sync>) -> Self {
53        Self { llm, tool_names: vec![] }
54    }
55
56    pub fn with_tools(mut self, tools: &[&str]) -> Self {
57        self.tool_names = tools.iter().map(|s| s.to_string()).collect();
58        self
59    }
60
61    pub async fn plan(&self, task: &str) -> Result<ExecutionPlan> {
62        let tools_str = if self.tool_names.is_empty() {
63            "none".to_string()
64        } else {
65            self.tool_names.join(", ")
66        };
67
68        let prompt = format!(
69            "You are a task planning assistant.\n\
70             Decompose the following task into concrete, sequential steps.\n\
71             If tools are available, indicate which tool to use for each step.\n\n\
72             Respond ONLY with a JSON array of objects with keys: \
73             \"step_number\", \"description\", \"tool\" (or null), \"expected_result\", \"dependencies\" (list of step numbers).\n\n\
74             Available tools: {tools}\n\n\
75             Task: {task}\n\n\
76             JSON plan:",
77            tools = tools_str,
78            task = task,
79        );
80
81        let messages = vec![
82            Message::system("You are a precise task planning assistant. Always respond with valid JSON."),
83            Message::user(&prompt),
84        ];
85
86        let response = self.llm.chat(&messages, &[]).await?;
87        let text = response.content.as_deref().unwrap_or("");
88
89        let steps = Self::parse_steps(text)?;
90        Ok(ExecutionPlan { goal: task.to_string(), steps })
91    }
92
93    fn parse_steps(raw: &str) -> Result<Vec<PlanStep>> {
94        // Try to find a JSON array
95        let start = raw.find('[').ok_or_else(|| anyhow!("No JSON array found in planner response"))?;
96        let end = raw.rfind(']').ok_or_else(|| anyhow!("No closing bracket in planner response"))? + 1;
97
98        let slice = &raw[start..end];
99        let parsed: Vec<PlanStep> = serde_json::from_str(slice)
100            .map_err(|e| anyhow!("Failed to parse plan steps: {e}"))?;
101
102        Ok(parsed)
103    }
104}
