1//! Local SQLite-backed vector store (no server needed).
2//!
3//! [`LocalChroma`] stores embeddings as f32 BLOBs in SQLite via `rusqlite`.
4//! Supports collection management, add/query/update/delete, cosine similarity
5//! computed in Rust. In-memory mode available via `:memory:` path.
6
7use anyhow::{Result, anyhow};
8use rusqlite::{params, Connection};
9use serde_json::Value;
10use std::path::Path;
11use std::sync::Mutex;
12use uuid::Uuid;
13
14use crate::core::embeddings::cosine_similarity;
15
16// ---------------------------------------------------------------------------
17// LocalChroma — persistent SQLite-backed vector store
18// ---------------------------------------------------------------------------
19pub struct LocalChroma {
20    conn: Mutex<Connection>,
21}
22
23#[derive(Debug, Clone)]
24pub struct LocalCollection {
25    pub id: String,
26    pub name: String,
27    pub metadata: Option<Value>,
28}
29
30#[derive(Debug, Clone)]
31pub struct LocalEntry {
32    pub id: String,
33    pub embedding: Option<Vec<f32>>,
34    pub metadata: Option<Value>,
35    pub document: Option<String>,
36}
37
38#[derive(Debug, Clone)]
39pub struct LocalQueryResult {
40    pub ids: Vec<String>,
41    pub distances: Vec<f64>,
42    pub documents: Vec<Option<String>>,
43    pub metadatas: Vec<Option<Value>>,
44}

46impl LocalChroma {
47    pub fn new(path: &str) -> Result<Self> {
48        let db_path = Path::new(path);
49        if let Some(parent) = db_path.parent() {
50            std::fs::create_dir_all(parent)?;
51        }
52        let conn = Connection::open(db_path)?;
53        let slf = Self { conn: Mutex::new(conn) };
54        slf.init_tables()?;
55        Ok(slf)
56    }
57
58    pub fn in_memory() -> Result<Self> {
59        let conn = Connection::open_in_memory()?;
60        let slf = Self { conn: Mutex::new(conn) };
61        slf.init_tables()?;
62        Ok(slf)
63    }
64
65    fn init_tables(&self) -> Result<()> {
66        let conn = self.conn.lock().unwrap();
67        conn.execute_batch(
68            "CREATE TABLE IF NOT EXISTS collections (
69                id    TEXT PRIMARY KEY,
70                name  TEXT UNIQUE NOT NULL,
71                metadata TEXT
72            );
73            CREATE TABLE IF NOT EXISTS entries (
74                id              TEXT PRIMARY KEY,
75                collection_id   TEXT NOT NULL REFERENCES collections(id) ON DELETE CASCADE,
76                embedding       BLOB,
77                metadata        TEXT,
78                document        TEXT
79            );
80            CREATE INDEX IF NOT EXISTS idx_entries_collection
81                ON entries(collection_id);",
82        )?;
83        Ok(())
84    }
85
86    fn blob_from_vec(v: &[f32]) -> Vec<u8> {
87        let bytes: Vec<u8> = v.iter()
88            .flat_map(|f| f.to_le_bytes())
89            .collect();
90        bytes
91    }
92
93    fn vec_from_blob(blob: &[u8]) -> Vec<f32> {
94        blob.chunks_exact(4)
95            .map(|c| f32::from_le_bytes([c[0], c[1], c[2], c[3]]))
96            .collect()
97    }

99    // -- collections --------------------------------------------------------
100    pub fn list_collections(&self) -> Result<Vec<LocalCollection>> {
101        let conn = self.conn.lock().unwrap();
102        let mut stmt = conn.prepare("SELECT id, name, metadata FROM collections ORDER BY id")?;
103        let rows = stmt.query_map([], |row| {
104            let id: String = row.get(0)?;
105            let name: String = row.get(1)?;
106            let meta_str: Option<String> = row.get(2)?;
107            let metadata = meta_str
108                .and_then(|s| serde_json::from_str(&s).ok());
109            Ok(LocalCollection { id, name, metadata })
110        })?;
111        rows.collect::<std::result::Result<Vec<_>, _>>()
112            .map_err(|e| anyhow!("{e}"))
113    }

115    pub fn get_collection(&self, name: &str) -> Result<LocalCollection> {
116        let conn = self.conn.lock().unwrap();
117        conn.query_row(
118            "SELECT id, name, metadata FROM collections WHERE name = ?1",
119            params![name],
120            |row| {
121                let id: String = row.get(0)?;
122                let name: String = row.get(1)?;
123                let meta_str: Option<String> = row.get(2)?;
124                let metadata = meta_str.and_then(|s| serde_json::from_str(&s).ok());
125        Ok(LocalCollection { id, name: name.to_string(), metadata })
126            },
127        ).map_err(|e| anyhow!("Collection '{name}' not found: {e}"))
128    }
129
130    pub fn create_collection(&self, name: &str, metadata: Option<Value>) -> Result<LocalCollection> {
131        let conn = self.conn.lock().unwrap();
132        let id = format!("col_{}", Uuid::new_v4().simple());
133        let meta_str = metadata.as_ref().map(|m| m.to_string());
134        conn.execute(
135            "INSERT INTO collections (id, name, metadata) VALUES (?1, ?2, ?3)",
136            params![id, name, meta_str],
137        )?;
138        Ok(LocalCollection { id, name: name.to_string(), metadata })
139    }
140
141    pub fn delete_collection(&self, name: &str) -> Result<()> {
142        let conn = self.conn.lock().unwrap();
143        let col = conn.query_row(
144            "SELECT id FROM collections WHERE name = ?1",
145            params![name],
146            |row| row.get::<_, String>(0),
147        ).map_err(|e| anyhow!("Collection '{name}' not found: {e}"))?;
148        conn.execute("DELETE FROM entries WHERE collection_id = ?1", params![col])?;
149        conn.execute("DELETE FROM collections WHERE id = ?1", params![col])?;
150        Ok(())
151    }

153    // -- add ----------------------------------------------------------------
154    pub fn add(
155        &self,
156        collection_id: &str,
157        ids: &[&str],
158        embeddings: Option<&[Vec<f32>]>,
159        metadatas: Option<&[Value]>,
160        documents: Option<&[&str]>,
161    ) -> Result<()> {
162        let conn = self.conn.lock().unwrap();
163        for (i, id) in ids.iter().enumerate() {
164            let emb_blob = embeddings
165                .and_then(|e| e.get(i))
166                .map(|v| Self::blob_from_vec(v));
167            let meta_str = metadatas
168                .and_then(|m| m.get(i))
169                .map(|v| v.to_string());
170            let doc = documents.and_then(|d| d.get(i)).map(|s| s.to_string());
171
172            conn.execute(
173                "INSERT INTO entries (id, collection_id, embedding, metadata, document)
174                 VALUES (?1, ?2, ?3, ?4, ?5)",
175                params![id, collection_id, emb_blob, meta_str, doc],
176            )?;
177        }
178        Ok(())
179    }

181    // -- query --------------------------------------------------------------
182    pub fn query(
183        &self,
184        collection_id: &str,
185        query_embeddings: &[Vec<f32>],
186        n_results: usize,
187        r#where: Option<Value>,
188    ) -> Result<LocalQueryResult> {
189        let conn = self.conn.lock().unwrap();
190        let q_emb = query_embeddings.first()
191            .ok_or_else(|| anyhow!("No query embeddings provided"))?;
192
193        let mut sql = String::from(
194            "SELECT id, embedding, metadata, document FROM entries WHERE collection_id = ?1"
195        );
196        let mut where_val: Option<String> = None;
197        if let Some(ref w) = r#where {
198            if let Some(obj) = w.as_object() {
199                for (field, val) in obj.iter().take(1) {
200                    if let Some(s) = val.as_str() {
201                        sql.push_str(&format!(" AND json_extract(metadata, '$.{field}') = ?2"));
202                        where_val = Some(s.to_string());
203                    }
204                }
205            }
206        }
207
208        let mut stmt = conn.prepare(&sql)?;
209        let rows: Vec<(String, Option<Vec<u8>>, Option<String>, Option<String>)> = {
210            let iter: Box<dyn Iterator<Item = _>> = if let Some(ref wv) = where_val {
211                Box::new(
212                    stmt.query_map(params![collection_id, wv], |row| {
213                        Ok((
214                            row.get::<_, String>(0)?,
215                            row.get::<_, Option<Vec<u8>>>(1)?,
216                            row.get::<_, Option<String>>(2)?,
217                            row.get::<_, Option<String>>(3)?,
218                        ))
219                    })?.filter_map(|r| r.ok())
220                )
221            } else {
222                Box::new(
223                    stmt.query_map(params![collection_id], |row| {
224                        Ok((
225                            row.get::<_, String>(0)?,
226                            row.get::<_, Option<Vec<u8>>>(1)?,
227                            row.get::<_, Option<String>>(2)?,
228                            row.get::<_, Option<String>>(3)?,
229                        ))
230                    })?.filter_map(|r| r.ok())
231                )
232            };
233            iter.collect()
234        };
235
236        let mut scored: Vec<(f64, String, Option<String>, Option<Value>)> = rows
237            .into_iter()
238            .filter_map(|(id, blob, meta_str, doc)| {
239                let emb = blob.as_deref().map(Self::vec_from_blob);
240                let dist = emb.as_ref()
241                    .map(|e| 1.0 - cosine_similarity(q_emb, e) as f64)
242                    .unwrap_or(f64::MAX);
243                let meta = meta_str.and_then(|s| serde_json::from_str(&s).ok());
244                if dist.is_finite() {
245                    Some((dist, id, doc, meta))
246                } else {
247                    None
248                }
249            })
250            .collect();
251
252        scored.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal));
253        scored.truncate(n_results);
254
255        let ids: Vec<String> = scored.iter().map(|(_, id, _, _)| id.clone()).collect();
256        let distances: Vec<f64> = scored.iter().map(|(d, _, _, _)| *d).collect();
257        let documents: Vec<Option<String>> = scored.iter().map(|(_, _, doc, _)| doc.clone()).collect();
258        let metadatas: Vec<Option<Value>> = scored.into_iter().map(|(_, _, _, meta)| meta).collect();
259
260        Ok(LocalQueryResult { ids, distances, documents, metadatas })
261    }

263    pub fn count(&self, collection_id: &str) -> Result<usize> {
264        let conn = self.conn.lock().unwrap();
265        let count: usize = conn.query_row(
266            "SELECT COUNT(*) FROM entries WHERE collection_id = ?1",
267            params![collection_id],
268            |row| row.get(0),
269        )?;
270        Ok(count)
271    }

273    pub fn delete_where(&self, collection_id: &str, r#where: Value) -> Result<()> {
274        let conn = self.conn.lock().unwrap();
275        if let Some(field) = r#where.as_object().and_then(|o| o.keys().next()) {
276            if let Some(val) = r#where.get(field).and_then(|v| v.as_str()) {
277                conn.execute(
278                    &format!(
279                        "DELETE FROM entries WHERE collection_id = ?1 AND json_extract(metadata, '$.{field}') = ?2"
280                    ),
281                    params![collection_id, val],
282                )?;
283            }
284        }
285        Ok(())
286    }

288    pub fn get_entries(&self, collection_id: &str) -> Result<Vec<LocalEntry>> {
289        let conn = self.conn.lock().unwrap();
290        let mut stmt = conn.prepare(
291            "SELECT id, embedding, metadata, document FROM entries WHERE collection_id = ?1"
292        )?;
293        let rows = stmt.query_map(params![collection_id], |row| {
294            let id: String = row.get(0)?;
295            let blob: Option<Vec<u8>> = row.get(1)?;
296            let meta_str: Option<String> = row.get(2)?;
297            let doc: Option<String> = row.get(3)?;
298            let embedding = blob.as_deref().map(Self::vec_from_blob);
299            let metadata = meta_str.and_then(|s| serde_json::from_str(&s).ok());
300            Ok(LocalEntry { id, embedding, metadata, document: doc })
301        })?;
302        rows.collect::<std::result::Result<Vec<_>, _>>()
303            .map_err(|e| anyhow!("{e}"))
304    }
