1//! Retrieval-Augmented Generation engine.
2//!
3//! [`RAGEngine`] ingests documents (txt, md, csv, pdf) or raw text → chunks via
4//! [`DocumentChunker`] → embeds via [`Embedding`]
5//! → stores in [`LocalChroma`] (SQLite-backed vector store).
6//! Query returns [`RAGResult`]s with text, source, metadata, and cosine distance.
7//!
8//! Also provides a [`ChromaDB`](chroma::ChromaDB) HTTP client for remote Chroma servers.
9
10use anyhow::Result;

23// ---------------------------------------------------------------------------
24// RAG config
25// ---------------------------------------------------------------------------
26pub struct RAGConfig {
27    pub chunk_size: usize,
28    pub chunk_overlap: usize,
29    pub default_k: usize,
30}
31
32impl Default for RAGConfig {
33    fn default() -> Self {
34        Self { chunk_size: 1000, chunk_overlap: 200, default_k: 5 }
35    }
36}

38// ---------------------------------------------------------------------------
39// RAG result
40// ---------------------------------------------------------------------------
41#[derive(Debug, Clone)]
42pub struct RAGResult {
43    pub text: String,
44    pub source: String,
45    pub metadata: Value,
46    pub distance: Option<f64>,
47}

49// ---------------------------------------------------------------------------
50// RAGEngine
51// ---------------------------------------------------------------------------
52pub struct RAGEngine {
53    pub store: LocalChroma,
54    pub embedding: Arc<dyn Embedding>,
55    pub config: RAGConfig,
56    chunker: DocumentChunker,
57}
58
59impl RAGEngine {
60    pub fn new(path: &str, embedding: Arc<dyn Embedding>, config: RAGConfig) -> Result<Self> {
61        let store = LocalChroma::new(path)?;
62        let chunker = DocumentChunker::new(config.chunk_size, config.chunk_overlap);
63        Ok(Self { store, embedding, config, chunker })
64    }
65
66    pub fn in_memory(embedding: Arc<dyn Embedding>, config: RAGConfig) -> Result<Self> {
67        let store = LocalChroma::in_memory()?;
68        let chunker = DocumentChunker::new(config.chunk_size, config.chunk_overlap);
69        Ok(Self { store, embedding, config, chunker })
70    }
71
72    pub fn collection_id(&self, name: &str) -> Result<String> {
73        let col = self.store.get_collection(name)?;
74        Ok(col.id)
75    }
76
77    pub fn create_collection(&self, name: &str) -> Result<String> {
78        let col = self.store.create_collection(name, None)?;
79        Ok(col.id)
80    }
81
82    // -- ingest file (txt, md, csv, pdf) ------------------------------------
83    pub async fn ingest_file(&self, collection_id: &str, file_path: &str) -> Result<()> {
84        let docs = DocumentLoader::load(file_path)?;
85        let chunks = self.chunker.chunk(&docs);
86        self.ingest_chunks(collection_id, &chunks).await
87    }
88
89    // -- ingest raw text ----------------------------------------------------
90    pub async fn ingest_text(&self, collection_id: &str, text: &str, metadata: Option<Value>) -> Result<()> {
91        let doc = Document::new(text, "raw_text");
92        let doc = match metadata {
93            Some(m) => Document { metadata: m, ..doc },
94            None => doc,
95        };
96        let chunks = self.chunker.chunk(&[doc]);
97        self.ingest_chunks(collection_id, &chunks).await
98    }
99
100    async fn ingest_chunks(&self, collection_id: &str, chunks: &[Document]) -> Result<()> {
101        if chunks.is_empty() {
102            return Ok(());
103        }
104
105        let texts: Vec<&str> = chunks.iter().map(|d| d.text.as_str()).collect();
106        let embeddings = self.embedding.embed_many(&texts).await?;
107
108        let ids: Vec<String> = (0..chunks.len())
109            .map(|_| format!("rag_{}", Uuid::new_v4().simple()))
110            .collect();
111        let ids_ref: Vec<&str> = ids.iter().map(|s| s.as_str()).collect();
112
113        let metadatas: Vec<Value> = chunks.iter().map(|d| d.metadata.clone()).collect();
114
115        let docs_ref: Vec<&str> = chunks.iter().map(|d| d.text.as_str()).collect();
116
117        self.store.add(
118            collection_id,
119            &ids_ref,
120            Some(&embeddings),
121            Some(&metadatas),
122            Some(&docs_ref),
123        )
124    }
125
126    // -- query --------------------------------------------------------------
127    pub async fn query(&self, collection_id: &str, query: &str, k: Option<usize>) -> Result<Vec<RAGResult>> {
128        let n = k.unwrap_or(self.config.default_k);
129        let query_emb = self.embedding.embed(query).await?;
130        let result = self.store.query(collection_id, &[query_emb], n, None)?;
131
132        let mut rag_results = Vec::new();
133
134        for i in 0..result.ids.len() {
135            let text = result.documents.get(i).cloned().flatten().unwrap_or_default();
136            let distance = result.distances.get(i).copied();
137            let metadata = result.metadatas.get(i).cloned().flatten().unwrap_or(Value::Object(Default::default()));
138            let source = metadata["source"].as_str().unwrap_or("unknown").to_string();
139
140            rag_results.push(RAGResult {
141                text,
142                source,
143                metadata,
144                distance,
145            });
146        }
147
148        Ok(rag_results)
149    }
150
151    pub fn delete_collection(&self, name: &str) -> Result<()> {
152        self.store.delete_collection(name)
153    }
154}
