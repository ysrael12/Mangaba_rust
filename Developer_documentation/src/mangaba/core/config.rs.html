1    //! Environment-based provider detection and [`LLMConfig`] builder.
2//!
3//! Mirrors the Python `Config` class: loads `.env` via `dotenvy`, detects the LLM provider
4//! from env vars, normalizes aliases (`gemini` → `google`, `gpt` → `openai`), and looks up
5//! provider-specific API keys with sensible defaults for model, temperature, and max tokens.
6
7use std::sync::OnceLock;
8
9use crate::core::types::LLMConfig;
10
11static LOAD_ENV: OnceLock<()> = OnceLock::new();
12
13fn ensure_env_loaded() {
14    LOAD_ENV.get_or_init(|| {
15        let _ = dotenvy::dotenv();
16    });
17}
18
19const SUPPORTED_PROVIDERS: &[&str] = &["google", "openai", "anthropic", "huggingface"];
20
21fn normalize_provider(raw: &str) -> Option<String> {
22    let p = raw.to_lowercase().replace('_', "-");
23    let normalized = match p.as_str() {
24        "gemini" | "google-ai" | "googleai" => "google",
25        "gpt" | "chatgpt" => "openai",
26        "claude" => "anthropic",
27        "hf" | "hugging-face" => "huggingface",
28        other => other,
29    };
30    if SUPPORTED_PROVIDERS.contains(&normalized) {
31        Some(normalized.to_string())
32    } else {
33        None
34    }
35}
36
37fn provider_env_keys() -> Vec<(&'static str, Vec<&'static str>)> {
38    vec![
39        ("google", vec!["GOOGLE_API_KEY", "GEMINI_API_KEY"]),
40        ("openai", vec!["OPENAI_API_KEY"]),
41        ("anthropic", vec!["ANTHROPIC_API_KEY"]),
42        ("huggingface", vec!["HUGGINGFACE_API_KEY", "HUGGINGFACE_TOKEN", "HF_TOKEN", "HUGGINGFACEHUB_API_TOKEN"]),
43    ]
44}
45
46fn default_model(provider: &str) -> &str {
47    match provider {
48        "google" => "gemini-2.5-flash",
49        "openai" => "gpt-4o-mini",
50        "anthropic" => "claude-3-haiku-20240307",
51        "huggingface" => "mistralai/Mistral-7B-Instruct-v0.2",
52        _ => "gemini-2.5-flash",
53    }
54}
55
56fn default_temperature(provider: &str) -> f32 {
57    let _ = provider;
58    0.7
59}

61fn env(name: &str) -> Option<String> {
62    std::env::var(name).ok().filter(|s| !s.is_empty())
63}

65pub struct Config {
66    pub provider: String,
67    pub model: String,
68    pub api_key: String,
69    pub temperature: f32,
70    pub max_tokens: usize,
71    pub base_url: Option<String>,
72    pub log_level: String,
73}
74
75impl Config {
76    pub fn load() -> Result<Self, String> {
77        ensure_env_loaded();
78
79        let provider = {
80            let raw = env("LLM_PROVIDER")
81                .or_else(|| env("AI_PROVIDER"))
82                .or_else(|| env("PROVIDER"))
83                .unwrap_or_else(|| "google".to_string());
84            normalize_provider(&raw).ok_or_else(|| {
85                format!(
86                    "Unsupported provider '{raw}'. Use one of: {}",
87                    SUPPORTED_PROVIDERS.join(", ")
88                )
89            })?
90        };
91
92        let model = env("MODEL_NAME")
93            .or_else(|| env("MODEL"))
94            .unwrap_or_else(|| default_model(&provider).to_string());
95
96        let key_vars = match provider_env_keys().into_iter().find(|(p, _)| p == &provider) {
97            Some((_, keys)) => keys,
98            None => vec![],
99        };
100
101        let api_key = key_vars.iter()
102            .find_map(|k| env(k))
103            .or_else(|| env("API_KEY"))
104            .ok_or_else(|| format!(
105                "No API key found for provider '{provider}'. Set one of: {}",
106                key_vars.iter().map(|k| k.to_string()).collect::<Vec<_>>().join(", ")
107            ))?;
108
109        let temperature = env("MODEL_TEMPERATURE")
110            .or_else(|| env("TEMPERATURE"))
111            .and_then(|v| v.parse::<f32>().ok())
112            .unwrap_or_else(|| default_temperature(&provider));
113
114        let max_tokens = env("MAX_OUTPUT_TOKENS")
115            .or_else(|| env("MAX_TOKENS"))
116            .and_then(|v| v.parse::<usize>().ok())
117            .unwrap_or(1024);
118
119        let base_url = env("BASE_URL");
120        let log_level = env("LOG_LEVEL").unwrap_or_else(|| "INFO".to_string());
121
122        Ok(Self { provider, model, api_key, temperature, max_tokens, base_url, log_level })
123    }
124
125    pub fn to_llm_config(&self) -> LLMConfig {
126        LLMConfig {
127            provider: self.provider.clone(),
128            model: self.model.clone(),
129            api_key: Some(self.api_key.clone()),
130            temperature: self.temperature,
131            max_tokens: self.max_tokens,
132            top_p: 1.0,
133            stop_sequences: None,
134            timeout: 60,
135            base_url: self.base_url.clone(),
136        }
137    }
138}
139
140impl std::fmt::Display for Config {
141    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
142        write!(f, "Config(provider={}, model={}, log_level={})", self.provider, self.model, self.log_level)
143    }
144}

146pub fn create_llm_config() -> LLMConfig {
147    match Config::load() {
148        Ok(cfg) => cfg.to_llm_config(),
149        Err(e) => {
150            log::warn!("Config load failed (using defaults): {e}");
151            LLMConfig {
152                provider: "google".into(),
153                model: "gemini-2.5-flash".into(),
154                api_key: None,
155                temperature: 0.7,
156                max_tokens: 1024,
157                top_p: 1.0,
158                stop_sequences: None,
159                timeout: 60,
160                base_url: None,
161            }
162        }
163    }
164}

166/// Provider detection helper — returns the provider from env or a default
167pub fn detect_provider() -> String {
168    ensure_env_loaded();
169    let raw = env("LLM_PROVIDER")
170        .or_else(|| env("AI_PROVIDER"))
171        .or_else(|| env("PROVIDER"))
172        .unwrap_or_else(|| "google".to_string());
173    normalize_provider(&raw).unwrap_or_else(|| "google".to_string())
174}
