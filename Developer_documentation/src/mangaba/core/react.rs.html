1//! ReAct (Reasoning + Acting) engine for LLM agents.
2//!
3//! [`ReActEngine`] implements the Thought → Action → Observation loop:
4//! 1. LLM generates a thought (and optionally tool calls)
5//! 2. Tools are executed and results appended as observations
6//! 3. Loop continues until the LLM produces a ﬁnal answer (no tool calls)
7//!    or `max_iterations` is reached.
8//!
9//! Each iteration ﬁres callbacks and events for observability.
10
11use anyhow::{Result, anyhow};
12use crate::core::{llm::LLMClient, tools::BaseTool};
13use crate::core::types::{LLMResponse, ReActStep, Message, ToolCall, ToolResult};
14use crate::core::events::{EventBus, Event, EventType};
15use crate::core::callbacks::Callbacks;
16
17pub struct ReActEngine<'a> {
18    pub llm: &'a dyn LLMClient,
19    pub tools: &'a [Box<dyn BaseTool + Send + Sync>],
20    pub callbacks: &'a Callbacks,
21    pub max_iterations: usize,
22    pub verbose: bool,
23}
24
25impl<'a> ReActEngine<'a> {
26    pub fn new(
27        llm: &'a dyn LLMClient,
28        tools: &'a [Box<dyn BaseTool + Send + Sync>],
29        callbacks: &'a Callbacks,
30        max_iterations: usize,
31        verbose: bool,
32    ) -> Self {
33        Self { llm, tools, callbacks, max_iterations, verbose }
34    }
35
36    pub async fn run(
37        &self,
38        messages: &mut Vec<Message>,
39    ) -> Result<(LLMResponse, Vec<ReActStep>)> {
40        let tool_refs: Vec<&dyn BaseTool> = self.tools.iter().map(|b| b.as_ref() as &dyn BaseTool).collect();
41        let mut steps = Vec::new();
42
43        for iteration in 0..self.max_iterations {
44            if self.verbose {
45                log::info!("ReAct iteration {}/{}", iteration + 1, self.max_iterations);
46            }
47
48            self.callbacks.call_llm_start(messages.len(), tool_refs.len());
49            let response = self.llm.chat(messages, &tool_refs).await?;
50            self.callbacks.call_llm_end(&response);
51
52            if let Some(ref thought) = response.content {
53                EventBus::emit(Event::new(
54                    EventType::ReActThought, "react",
55                    serde_json::json!({"step": iteration + 1, "thought": thought}),
56                ));
57            }
58
59            let mut step = ReActStep {
60                step_number: iteration + 1,
61                thought: response.content.clone(),
62                action: None,
63                observation: None,
64                timestamp: chrono::Utc::now().to_rfc3339(),
65            };
66
67            if response.has_tool_calls() {
68                EventBus::emit(Event::new(
69                    EventType::ReActAction, "react",
70                    serde_json::json!({"step": iteration + 1, "tool_calls": response.tool_calls}),
71                ));
72
73                messages.push(Message::assistant(
74                    response.content.clone(),
75                    Some(response.tool_calls.clone()),
76                ));
77
78                for tool_call in &response.tool_calls {
79                    self.callbacks.call_tool_start(&tool_call.tool_name, &serde_json::json!(tool_call.arguments));
80                    let result = self.execute_tool(tool_call, iteration).await;
81                    self.callbacks.call_tool_end(&tool_call.tool_name, &result);
82
83                    step.action = Some(tool_call.clone());
84                    step.observation = result.output.clone()
85                        .map(|o| o.to_string())
86                        .or_else(|| result.error.clone());
87
88                    messages.push(Message::tool(vec![result]));
89
90                    EventBus::emit(Event::new(
91                        EventType::ReActObservation, "react",
92                        serde_json::json!({"step": iteration + 1, "observation": step.observation}),
93                    ));
94
95                    if let Some(tool) = self.tools.iter().find(|t| t.name() == tool_call.tool_name) {
96                        if tool.return_direct() {
97                            self.callbacks.call_step(&step, "");
98                            steps.push(step);
99                            return Ok((response, steps));
100                        }
101                    }
102                }
103            } else {
104                messages.push(Message::assistant(response.content.clone(), None));
105                self.callbacks.call_step(&step, "");
106                steps.push(step);
107                return Ok((response, steps));
108            }
109
110            self.callbacks.call_step(&step, "");
111            steps.push(step);
112        }
113
114        Err(anyhow!("ReAct loop exceeded max_iterations ({})", self.max_iterations))
115    }
116
117    async fn execute_tool(&self, tool_call: &ToolCall, _step: usize) -> ToolResult {
118        let tool = match self.tools.iter().find(|t| t.name() == tool_call.tool_name) {
119            Some(t) => t,
120            None => {
121                return ToolResult {
122                    call_id: tool_call.id.clone(),
123                    tool_name: tool_call.tool_name.clone(),
124                    output: None,
125                    error: Some(format!("Tool '{}' not found", tool_call.tool_name)),
126                    success: false,
127                };
128            }
129        };
130
131        let args_value = serde_json::json!(tool_call.arguments);
132        match tool.call(args_value).await {
133            Ok(result) => result,
134            Err(e) => ToolResult {
135                call_id: tool_call.id.clone(),
136                tool_name: tool_call.tool_name.clone(),
137                output: None,
138                error: Some(format!("{}", e)),
139                success: false,
140            },
141        }
142    }
143}
