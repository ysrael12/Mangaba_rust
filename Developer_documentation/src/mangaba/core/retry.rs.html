1//! Exponential backoff retry with jitter for fallible operations.
2//!
3//! [`RetryConfig`] conﬁgures max retries, initial/max backoff, and jitter factor.
4//! [`with_retry`] executes a closure, retrying on failure with increasing delays.
5
6use std::time::Duration;
7use rand::Rng;
8use crate::core::events::{EventBus, Event, EventType};
9use crate::core::errors::MangabaError;
10
11#[derive(Debug, Clone)]
12pub struct RetryConfig {
13    pub max_retries: u32,
14    pub initial_backoff_ms: u64,
15    pub max_backoff_ms: u64,
16    pub jitter_factor: f64,
17}
18
19impl Default for RetryConfig {
20    fn default() -> Self {
21        Self {
22            max_retries: 3,
23            initial_backoff_ms: 1000,
24            max_backoff_ms: 30_000,
25            jitter_factor: 0.25,
26        }
27    }
28}

30/// Classify an error into `(should_retry, forced_wait_ms)`.
31///
32/// This is where the [`MangabaError`] taxonomy finally drives behavior:
33/// - `RateLimit { retry_after }` → retry, but honor the server-provided delay
34///   instead of our own backoff.
35/// - any other [`MangabaError`] → retry only if `is_retryable()` (so auth
36///   failures, validation errors and malformed-JSON errors fail fast).
37/// - non-`MangabaError` (network/timeouts from `reqwest`, etc.) → treated as
38///   transient and retried.
39fn classify(err: &anyhow::Error) -> (bool, Option<u64>) {
40    match err.downcast_ref::<MangabaError>() {
41        Some(MangabaError::RateLimit { retry_after, .. }) => (true, Some(retry_after.saturating_mul(1000))),
42        Some(other) => (other.is_retryable(), None),
43        None => (true, None),
44    }
45}

47pub async fn with_retry<T, F, Fut>(config: &RetryConfig, source_id: &str, operation_name: &str, f: F) -> anyhow::Result<T>
48where
49    F: Fn() -> Fut,
50    Fut: std::future::Future<Output = anyhow::Result<T>>,
51{
52    let max_attempts = config.max_retries.max(1);
53
54    for attempt in 1..=max_attempts {
55        match f().await {
56            Ok(val) => return Ok(val),
57            Err(e) => {
58                let (retryable, forced_wait_ms) = classify(&e);
59
60                // Fail fast on non-retryable errors (auth, validation, parse) —
61                // burning the retry budget on them only adds latency.
62                if !retryable || attempt >= max_attempts {
63                    return Err(e);
64                }
65
66                EventBus::emit(Event::new(
67                    EventType::LLMRetry,
68                    source_id,
69                    serde_json::json!({
70                        "operation": operation_name,
71                        "attempt": attempt,
72                        "max_retries": max_attempts,
73                        "error": format!("{}", e),
74                    }),
75                ));
76
77                let backoff = forced_wait_ms.unwrap_or_else(|| compute_backoff(config, attempt as u64));
78                tokio::time::sleep(Duration::from_millis(backoff)).await;
79            }
80        }
81    }
82
83    // Unreachable: the loop either returns Ok or returns the last Err above.
84    Err(anyhow::anyhow!("Operation '{}' exhausted retries", operation_name))
85}
86
87fn compute_backoff(config: &RetryConfig, attempt: u64) -> u64 {
88    let exponential = config.initial_backoff_ms * 2u64.saturating_pow(attempt as u32 - 1);
89    let capped = exponential.min(config.max_backoff_ms);
90    let jitter_range = (capped as f64 * config.jitter_factor) as u64;
91    let mut rng = rand::thread_rng();
92    let jitter = if jitter_range > 0 { rng.gen_range(0..=jitter_range) } else { 0 };
93    capped.saturating_add(jitter).min(config.max_backoff_ms)
94}

96#[cfg(test)]
97mod tests {
98    use super::*;
99    use std::sync::atomic::{AtomicUsize, Ordering};
100    use std::sync::Arc;
101
102    fn fast_config() -> RetryConfig {
103        RetryConfig { max_retries: 5, initial_backoff_ms: 1, max_backoff_ms: 4, jitter_factor: 0.0 }
104    }
105
106    #[test]
107    fn backoff_is_exponential_and_capped() {
108        let cfg = RetryConfig { max_retries: 10, initial_backoff_ms: 100, max_backoff_ms: 500, jitter_factor: 0.0 };
109        assert_eq!(compute_backoff(&cfg, 1), 100);
110        assert_eq!(compute_backoff(&cfg, 2), 200);
111        assert_eq!(compute_backoff(&cfg, 3), 400);
112        assert_eq!(compute_backoff(&cfg, 4), 500); // capped
113        assert_eq!(compute_backoff(&cfg, 50), 500); // saturating, still capped
114    }
115
116    #[test]
117    fn classify_distinguishes_fatal_from_transient() {
118        let auth: anyhow::Error = MangabaError::Authentication("bad key".into()).into();
119        assert_eq!(classify(&auth), (false, None));
120
121        let rate: anyhow::Error = MangabaError::RateLimit { retry_after: 3, detail: "slow down".into() }.into();
122        assert_eq!(classify(&rate), (true, Some(3000)));
123
124        let network = anyhow::anyhow!("connection reset");
125        assert_eq!(classify(&network), (true, None));
126    }
127
128    #[tokio::test]
129    async fn retries_transient_then_succeeds() {
130        let calls = Arc::new(AtomicUsize::new(0));
131        let c = calls.clone();
132        let cfg = fast_config();
133        let result: anyhow::Result<u32> = with_retry(&cfg, "test", "op", || {
134            let c = c.clone();
135            async move {
136                let n = c.fetch_add(1, Ordering::SeqCst);
137                if n < 2 { Err(anyhow::anyhow!("transient network error")) } else { Ok(42) }
138            }
139        }).await;
140        assert_eq!(result.unwrap(), 42);
141        assert_eq!(calls.load(Ordering::SeqCst), 3);
142    }

144    #[tokio::test]
145    async fn fatal_error_is_not_retried() {
146        let calls = Arc::new(AtomicUsize::new(0));
147        let c = calls.clone();
148        let cfg = fast_config();

156        assert!(result.is_err());
157        // Auth errors must fail on the first attempt — no wasted retries.
